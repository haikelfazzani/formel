# Example

 /* default css code */
.split-view {
  width: 100%;
  display: flex;
}

.gutter { 
  flex: 0 0 10px;
}

.split-view.horizontal {
  flex-direction: row;
}
.split-view.vertical {
  flex-direction: column;
}

.split-view.horizontal>.gutter {
  cursor: col-resize;
}

.split-view.vertical>.gutter {
  cursor: row-resize;
}
 // js code
const options = {
  parent: '.split-view', // or document.querySelector('.split-view')
  direction: 'horizontal',
  gutterSize: 5,
  minSize: 20,
  sizes: [25, 50, 25],
  onDragEnd: (newSizes) => {
    console.log(newSizes);
  }
};

SplitViews(options);
