.split-view {
  width: 100%;
  height: 400px;
  /*   
    display: flex; (required)
    SplitViews is flexbox based
  */
  display: flex;
}

.horizontal { 
  flex-direction: row; 
}

.vertical { 
  flex-direction: colmun; 
}
import SplitViews from 'split-views';

SplitViews({
  parent: '.split-view',
  direction: 'horizontal',
  gutterSize: 10,
  minSize: 0,
  sizes: [25, 50, 25],
  onDragEnd: (newSizes) => {
    console.log(newSizes);
  }
});

// Vertical direction
window.SplitViews({
  parent: '.split-view2',
  direction: 'vertical',
  gutterSize: 5,
  minSize: 20,
  sizes: [75, 25]
});
